1. Crear el proyecto con nest
    pnpm dlx @nestjs/cli new encomiendas

2. Instalar dependencias
    pnpm add @nestjs/typeorm typeorm pg
    pnpm add @nestjs/config

3. Crear la base de datos
    crear el archivo .env
    DB_HOST=localhost
    DB_PORT=5432
    DB_USERNAME=postgres
    DB_PASSWORD=123456
    DB_NAME=encomiendas
    modificar el archivo app.module.ts
    con configModule y TypeOrmModule
    
    imports: [
    ConfigModule.forRoot({
      isGlobal: true,
    }),
    TypeOrmModule.forRootAsync({
      inject: [ConfigService],
      useFactory: (config: ConfigService) => ({
        type: 'postgres',
        host: config.get<string>('DB_HOST'),
        port: config.get<number>('DB_PORT'),
        username: config.get<string>('DB_USER'),
        password: config.get<string>('DB_PASS'),
        database: config.get<string>('DB_NAME'),
        autoLoadEntities: true,
        synchronize: true,
      }),
    }),
    UsersModule,
  ],

4. Crear el modelo de usuario
    crear el archivo user.entity.ts
    import { Entity, PrimaryGeneratedColumn, Column, CreateDateColumn } from 'typeorm';
    
    @Entity()
    export class User {
      @PrimaryGeneratedColumn()
      id: number;
      
      @Column()
      email: string;
      
      @Column()
      password: string;
      
      @CreateDateColumn()
      createdAt: Date;
    }
    
5. registrar entidad en el modulo
    modificar el archivo users.module.ts
    con TypeOrmModule.forFeature([User])

7. conectar repositorio en ConfigService
    @Injectable()
    export class UsersService {
      // Constructor que inyecta el repositorio de usuarios
      constructor(
        @InjectRepository(User)
        private userRepository: Repository<User>
      ) {}
    
      // Crear usuario
      create(createUserDto: CreateUserDto) {
        return this.userRepository.save(createUserDto);
      }
    
      // Obtener todos los usuarios
      findAll() {
        return this.userRepository.find();
      }
    
      findOne(id: number) {
        return this.userRepository.findOneBy({ id });
      }
    
      update(id: number, updateUserDto: UpdateUserDto) {
        return this.userRepository.update(id, updateUserDto);
      }
    
      remove(id: number) {
        return this.userRepository.delete(id);
      }
    }

8. conectar el controller con el service
    @Controller('users')
    export class UsersController {
      constructor(private readonly usersService: UsersService) {}

      @Post()
      create(@Body() createUserDto: CreateUserDto) {
        return this.usersService.create(createUserDto);
      }

      @Get()
      findAll() {
        return this.usersService.findAll();
      }

      @Get(':id')
      findOne(@Param('id') id: string) {
        return this.usersService.findOne(+id);
      }

      @Patch(':id')
      update(@Param('id') id: string, @Body() updateUserDto: UpdateUserDto) {
        return this.usersService.update(+id, updateUserDto);
      }

      @Delete(':id')
      remove(@Param('id') id: string) {
        return this.usersService.remove(+id);
      }
    }

9. conectar modulo
    @Module({
      imports: [TypeOrmModule.forFeature([User])],
      controllers: [UsersController],
      providers: [UsersService],
    })
    export class UsersModule {}
